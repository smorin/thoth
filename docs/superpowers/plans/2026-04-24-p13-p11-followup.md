```python
def is_background_model(model: str | None) -> bool:
    """Return True if a model name implies background/long-running submission.

    Rule: any model name containing the substring "deep-research" is treated
    as background. Case-sensitive by design (OpenAI model IDs are lowercase).
    `None` and empty string return False.
    """
    return "deep-research" in (model or "")


def is_background_mode(mode_config: dict[str, Any]) -> bool:
    """Return True if a mode submits as a long-running background job.

    Precedence: explicit `async` key wins (truthy → background, falsy →
    immediate); otherwise delegate to `is_background_model` using the model
    name.
    """
    if "async" in mode_config:
        return bool(mode_config["async"])
    return is_background_model(mode_config.get("model"))
```

```python
"""Shared secret-masking helpers used by config_cmd and modes_cmd.

Rule: any dict key whose final dotted-path segment is `api_key` is masked —
the value is replaced with `****<last-four-chars>`. Env-template placeholders
like `${OPENAI_API_KEY}` pass through unchanged.
"""

from __future__ import annotations

from typing import Any

_SECRET_KEY_SUFFIX = "api_key"


def _is_secret_key(key: str) -> bool:
    return key.split(".")[-1] == _SECRET_KEY_SUFFIX


def _mask_secret(value: Any) -> Any:
    if value is None:
        return None
    if not isinstance(value, str) or not value:
        return value
    if value.startswith("${") and value.endswith("}"):
        return value
    tail = value[-4:] if len(value) >= 4 else value
    return f"****{tail}"


def _mask_tree(data: Any, prefix: str = "") -> Any:
    if isinstance(data, dict):
        return {
            k: _mask_tree(v, f"{prefix}.{k}" if prefix else k) for k, v in data.items()
        }
    if isinstance(data, list):
        return [_mask_tree(v, prefix) for v in data]
    if prefix and _is_secret_key(prefix):
        return _mask_secret(data)
    return data


__all__ = ["_is_secret_key", "_mask_secret", "_mask_tree"]
```

Note on semantics: the existing `config_cmd._mask_secret` short-circuited on empty string via `not value`. The shared version does the same (`not value` returns the value unchanged). For `None`, the existing `modes_cmd._mask_secret` returned `None` (because `isinstance(None, str)` is False, it returned the value unchanged). The shared version makes that explicit with the `if value is None: return None` branch — same observable behaviour.
